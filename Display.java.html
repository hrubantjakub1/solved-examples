package com.view; 
 
import javax.imageio.ImageIO; 
import javax.swing.*; 
import java.awt.*; 
import java.awt.image.BufferedImage; 
import java.io.File; 
import java.io.IOException; 
 
/** 
 * Created by kubikulas on 4.6.17. 
 */ 
/* 
this method creates window where chart is displaied, it display chart 
*/ 
public class Display { 
 
    public Display() throws IOException { 
        BufferedImage img = ImageIO.read(new File("XYLineChart.jpeg")); 
        ImageIcon icon = new ImageIcon(img); 
        JFrame frame = new JFrame(); 
        frame.setLayout(new FlowLayout()); 
        frame.setSize(700, 500); 
        JLabel lbl = new JLabel(); 
        lbl.setIcon(icon); 
        frame.add(lbl); 
        frame.setVisible(true); 
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); 
 
        // This method creates window where chart is displayed 
    } 
}
