public class Morg { 
    private double moBirth; 
    private double moClear; 
 
    public Morg(double moBirth, double moClear) { 
        if ((moBirth >= 0 && moBirth <= 1)) { 
            setMoBirth(moBirth); 
            if (moClear >= 0 && moClear <= 1) { 
                setMoClear(moClear); 
            } else throw new AssertionError(); 
        } else throw new AssertionError(); 
 
        // Initializating the value of the values moBirth, moClear and stores them as numeric values 
        // check if the variables is greater than or equal to 0 and less or equal to 1 
    } 
 
    public Morg() { 
    } 
 
    public String toString() { 
        return "main.java.com.base.Morg: (" + moBirth + ", " + moClear + ")"; 
    } 
 
    public double getMoBirth() { 
        return moBirth; 
    } 
 
    public void setMoBirth(double moBirth) { 
        this.moBirth = moBirth; 
    } 
 
    public double getMoClear() { 
        return moClear; 
    } 
 
    public void setMoClear(double moClear) { 
        this.moClear = moClear; 
    } 
 
    public boolean moClear() { 
 
        return Math.random() < getMoClear(); 
 
        // clear Morg Object if the random number is greater than getMoClear() 
    } 
 
    public Morg moBirthNew(double moDens) { 
 
        if (Math.random() < moBirth * (1 - moDens)) { 
            // find out if a new microorganism (Morg) births, 
            // other microorganisms are born with less probability how the population grows 
            return new Morg(getMoBirth(), moDens); 
        } else { 
            return null; 
            // returns null if the microorganism does not multiply 
        } 
    } 
}
