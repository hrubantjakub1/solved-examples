package com.base;

public class Morg { 
    private double moBirth; 
    private double moClear; 
 
    public Morg(double moBirth, double moClear) { 
        if ((moBirth >= 0 && moBirth <= 1)) { 
            setMoBirth(moBirth); 
            if (moClear >= 0 && moClear <= 1) { 
                setMoClear(moClear); 
            } else throw new AssertionError(); 
        } else throw new AssertionError(); 
 
        // Initialization of microorganism values moBirth, moClear and saves them like numeric variable 
        // check whether is variable more or equal than 0 and less or equal than 1 
    } 
 
    public Morg() { 
    } 
 
    public String toString() { 
        return "main.java.com.base.Morg: (" + moBirth + ", " + moClear + ")"; 
    } 
 
    public double getMoBirth() { 
        return moBirth; 
    } 
 
    public void setMoBirth(double moBirth) { 
        this.moBirth = moBirth; 
    } 
 
    public double getMoClear() { 
        return moClear; 
    } 
 
    public void setMoClear(double moClear) { 
        this.moClear = moClear; 
    } 
 
    public boolean moClear() { 
 
        return Math.random() < getMoClear(); 
 
        // clear morg if random number is greater than getMoClear() 
    } 
 
    public Morg moBirthNew(double moDens) { 
 
        if (Math.random() < moBirth * (1 - moDens)) { 
            return new Morg(getMoBirth(), moDens); 
        } else { 
            return null; 
        } 
 
        // find out whether new microorganism births, 
        // next morg births with smaller probability as population is greater and greater 
        // moDens is morg population divided by the maximum population, 
        // returns null if morg does not reproduce 
    } 
}
