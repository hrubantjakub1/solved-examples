package com.base; 
 
import java.util.ArrayList; 
import java.util.List; 
 
/** 
 * Created by kubikulas on 4.6.17. 
 */ 
 
 
public class Body extends Morg { 
    private int max_morg_number; 
    private List<Body> morgs_list; 
    private Morg m; 
    private double bMorg; 
    private double cMorg; 
 
    // max_morg_number: the  maximum allowed microorganism population for this human body 
    // bMorg, birth morg probability, inherits value from moBirth 
    // cMorg, clear morg probability, inherits value from moClear 
    // m is microorganism, Object, that contains moBirth and moClear 
    //morgs_list is list of microorganisms 
 
    public Body(double bMorg, double cMorg) { 
        super(bMorg, cMorg); 
    } 
 
    public Body(List<Body> morgs_list, int max_morg_number) { 
        this.morgs_list = morgs_list; 
        this.max_morg_number = max_morg_number; 
    } 
 
    public Body() { 
    } 
 
    public String toString() { 
        return "main.java.com.base.Body: (" + getbMorg() + "," + getcMorg() + ")"; 
        //prints body object 
    } 
 
    public double getbMorg() { 
        return super.getMoBirth(); 
    } 
 
    public void setbMorg(double bMorg) { 
        this.bMorg = bMorg; 
    } 
 
    public double getcMorg() { 
        return super.getMoClear(); 
    } 
 
    public void setcMorg(double cMorg) { 
        this.cMorg = cMorg; 
    } 
 
    public int getPop() { 
 
        //returns: The total virus population (an integer) 
 
        return morgs_list.size(); 
    } 
 
    public int getMax_morg_number() { 
        return max_morg_number; 
    } 
 
    public void setMax_morg_number(int max_morg_number) { 
        this.max_morg_number = max_morg_number; 
    } 
 
    public List<Body> getMorgs_list() { 
        return morgs_list; 
    } 
 
    public void setMorgs_list(List<Body> morgs_list) { 
        this.morgs_list = morgs_list; 
    } 
 
    public void getCurrentDensity() { 
 
        // getCurrentDensity updates morgs population 
        // check whether morg survives, then calculates population density 
        // find out whether morg birth in moBirthNew and add morg to the list (morgs_list) 
 
        List<Body> alive_morgs_list = new ArrayList<>(); 
        for (int i = 0; i < morgs_list.size(); i++) { 
            if (!(morgs_list.get(i).moClear())) { 
                alive_morgs_list.add(morgs_list.get(i)); 
            } 
          // check whether morg survives 
        } 
 
        morgs_list = alive_morgs_list; 
        double current_density = getPop() / max_morg_number; 
        // calculates population density like total Pop divided by max morg population 
        List<Body> morgs_c = new ArrayList<>(); 
        for (Morg morg : morgs_list) { 
            m = morg.moBirthNew(current_density); 
            if (m != null) { 
                morgs_c.add((Body) morg); 
            } 
            // create variable m that stores new birth morg 
            // and check whether morg born or not (if variable m is not null) 
        } 
        morgs_list.addAll(morgs_c); 
 
        // add all born morgs to morgs_list 
    } 
}
